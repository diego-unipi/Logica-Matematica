\section{Sintassi}

\begin{definition}[Formula in $\mathsf{ND}$]
    Sia $L$ un linguaggio, una \vocab{$L$-formula in $\mathsf{ND}$} è una sequenza finita di simboli nell'alfabeto $\{\land, \lor, \to, \neg, \forall, \exists, \top, \bot\} \cup L$ (il primo insieme è costituito dai \vocab{simboli logici}).
\end{definition}

\begin{definition}[Dimostrazione in $\mathsf{ND}$]
    Una \vocab{dimostrazione in $\mathsf{ND}$} di una $L$-formula $\varphi$ è una sequenza finita di $L$-formule $\varphi_1, \varphi_2, \ldots, \varphi_n$ tale che $\varphi_n = \varphi$ e per ogni $1 \leq k \leq n$
    o $\varphi_k$ è un assioma, oppure $\varphi_k$ è in $T$, oppure $\varphi_k$ è ottenuta da formule precedenti tramite una \vocab{regola di inferenza} del \vocab{sistema di deduzione naturale} (\vocab{$\mathsf{ND}$}) qui di seguito riportate.
\end{definition}

\begin{definition}[Sequente in $\mathsf{ND}_{\to, \bot, \exists, =}$]
    Un \vocab{sequente} $T \vdash_{\mathsf{ND}_{\to, \bot, \exists, =}} \varphi$ indica che esiste una dimostrazione in $\mathsf{ND}_{\to, \bot, \exists, =}$ di $\varphi$ a partire dalle formule in $T$.
\end{definition}

\begin{definition}[$\textcolor{purple}{\neg, \top, \land, \lor, \forall}$]
    Definiamo i seguenti simboli logici nel sistema ridotto a partire da quelli di base:
    \begin{alignat*}{4}
        \textcolor{purple}{\neg}\,\varphi &\overset{\text{def}}{=} \varphi \,\textcolor{purple}{\to}\, \textcolor{purple}{\bot} \qquad & \textcolor{purple}{\top} &\overset{\text{def}}{=} \textcolor{purple}{\neg\bot} \qquad\quad & \varphi \,\textcolor{purple}{\land}\, \psi &\overset{\text{def}}{=} \neg(\varphi \,\textcolor{purple}{\to}\, \neg\psi) \\
        \varphi \,\textcolor{purple}{\lor}\, \psi &\overset{\text{def}}{=} \neg\varphi \,\textcolor{purple}{\to}\, \psi \qquad & \textcolor{purple}{\forall x_k}\,\varphi &\overset{\text{def}}{=} \textcolor{purple}{\neg\,\exists x_k\,\neg}\,\varphi
    \end{alignat*}
\end{definition}

\begin{exercise}[$\mathsf{ND}_{\to, \bot, \exists, =}$ dimostra $\mathsf{ND}$]
    In $\mathsf{ND}_{\to, \bot, \exists, =}$ si possono dimostrare le regole mancanti di $\mathsf{ND}$ nel seguente ordine: 
    \textcolor{green}{In}$_{\neg}$, \textcolor{orange}{El}$_{\bot}$, \textcolor{orange}{El}$_{\neg}$, \textcolor{green}{In}$_{\top}$, \textcolor{green}{In}$_{\land}$, \textcolor{orange}{El}$_{\land\cdot}$, \textcolor{orange}{El}$_{\cdot\land}$, \textcolor{green}{In}$_{\lor\cdot}$, \textcolor{green}{In}$_{\cdot\lor}$,
    \textcolor{orange}{El}$_{\lor}$, \textcolor{green}{In}$_{\forall}$, \textcolor{orange}{El}$_{\forall}$. 
\end{exercise}

\begin{exercise}[Semantica di Tarski indotta]
    Verificare che la semantica di Tarski, già definita in precedenza, coincide con quella indotta dalle definizioni date sopra 
    per i simboli: $\textcolor{purple}{\neg, \top, \land, \lor, \forall}$.\footnote{Nella pratica, ad esempio $M \models \{v\}\neg\varphi$ è $M \models \{v\}\varphi \to \bot$, a questo punto voglio vedere che ottengo la stessa formula insiemistica che ottenevo nella definizione di semantica di Tarski, per cui $\neg(M \models \{v\}\varphi)$ deve avere 
    lo stesso valore di verità (cioè si ha proprio $\iff$) di $M \models \{v\}\varphi \to M \models \{v\}\bot$ (ricordiamo che il $\to$ è quello metateorico in questo caso).}
\end{exercise}

\begin{notation}[Sequenti]
    Avendo verificato che $\mathsf{ND}$ e $\mathsf{ND}_{\to, \bot, \exists, =}$ sono equivalenti, d'ora in avanti indicheremo semplicemente con $T \vdash \varphi$ il sequente dimostrabile in entrambi i sistemi.
\end{notation}

\begin{definition}[Formule $T$-deduttivamente equivalenti]
    Due $L$-formule $\varphi$ e $\psi$ sono $T$-\vocab{deduttivamente equivalenti} (o \vocab{deduttivamente equivalenti} per $T$, o \vocab{interderivabili} per $T$) se $T \vdash \varphi \to \psi$ e $T \vdash \psi \to \varphi$. O equivalentemente se $T, \varphi \vdash \psi$ e $T, \psi \vdash \varphi$.
\end{definition}

\begin{remark}[Formule deduttivamente equivalenti]
    Segue quindi che due formule sono deduttivamente equivalenti (o interderivabili) se $\vdash \varphi \to \psi$ e $\vdash \psi \to \varphi$, o equivalentemente se $\varphi \vdash \psi$ e $\psi \vdash \varphi$.
\end{remark}

\begin{example}[Doppia negazione]
    Dimostriamo che $\varphi \vdash \neg\neg\varphi$ e $\neg\neg\varphi \vdash \varphi$ per ogni $L$-formula $\varphi$.
    \begin{align*}
        \text{In}_\to\frac{\displaystyle\text{In}_\neg\frac{\displaystyle\text{El}_\neg\frac{\displaystyle \text{Wk}\frac{\displaystyle \text{Ax}\frac{}{\varphi \vdash \varphi}}{\varphi,\neg\varphi\vdash \varphi} \; \text{Wk}\frac{\displaystyle\text{Ax}\frac{}{\neg \varphi \vdash \neg\varphi}}{\varphi,\neg\varphi\vdash \neg\varphi}}{\varphi,\neg\varphi\vdash\bot}}{\varphi \vdash \neg(\neg\varphi)}}{\vdash \varphi \to \neg(\neg \varphi)}
    \end{align*}
    Analogamente per $\neg\neg\varphi \vdash \varphi$.
    \begin{align*}
        \text{RaA}\frac{\displaystyle\text{El}_\neg\frac{\displaystyle\text{Wk}\frac{\displaystyle\text{Ax}\frac{}{\neg\neg\varphi \vdash \neg\neg\varphi}}{\neg\neg\varphi, \neg\varphi \vdash \neg\neg\varphi} \; \text{Wk}\frac{\displaystyle\text{Ax}\frac{}{\neg\varphi \vdash \neg\varphi}}{\neg\neg\varphi, \neg\varphi \vdash \neg\varphi}}{\neg\neg\varphi, \neg\varphi \vdash \bot}}{\neg\neg\varphi \vdash \varphi}
    \end{align*}
\end{example}

\begin{example}[Principio del terzo escluso]
    Dimostriamo che $\vdash \varphi \lor \neg\varphi$ per ogni $L$-formula $\varphi$.
    \begin{align*}
        \text{In}_\lor\cdot\frac{\displaystyle\text{In}_\neg\frac{\displaystyle\text{El}_\neg\frac{\displaystyle \text{Wk}\frac{\displaystyle \text{Ax}\frac{}{\varphi \vdash \varphi}}{\varphi,\neg\varphi\vdash \varphi} \; \text{Wk}\frac{\displaystyle\text{Ax}\frac{}{\neg \varphi \vdash \neg\varphi}}{\varphi,\neg\varphi\vdash \neg\varphi}}{\varphi,\neg\varphi\vdash\bot}}{\varphi \vdash \neg(\neg\varphi)}}{\vdash \varphi \lor \neg(\neg \varphi)}
    \end{align*}
    Quindi, fatto questo, d'ora in poi potremmo aggiungere una nuova regola ammissibile al nostro sistema deduttivo che ci dice:
    \[ \frac{}{\vdash \varphi \lor \neg\varphi} \]
    per ogni $L$-formula $\varphi$.
\end{example}

\subsection{Teoremi di correttezza e completezza}
Il nostro desiderio a questo punto è quello di dimostrare che $\models$ e $\vdash$ coincidono, cioè che $T \vdash \varphi$ se e solo se $T \models \varphi$. Per cui vorremmo dimostrare dei teoremi di correttezza e completezza, che ci dicono rispettivamente che $T \vdash \varphi$ implica $T \models \varphi$ e che $T \models \varphi$ implica $T \vdash \varphi$.
Tuttavia, per come abbiamo dato le definizioni c'è un problema.

\begin{example}[Esempio di conseguenza logica non dimostrabile]
    Consideriamo la seguente conseguenza logica nel linguaggio vuoto: $x_1 = x_2 \models x_3 = x_4$. Per via della semantica di Tarski, questa conseguenza logica si traduce in: per ogni $L$-struttura $M$ tale che $M \models x_1 = x_2$, allora $M \models x_3 = x_4$, i.e.:
    \[ \forall v : \Var \to M \; M \models \{v\}x_1 = x_2 \implies M \models \{v\}x_3 = x_4 \]
    cioè per ogni valutazione $v$, se $v(x_1) = v(x_2)$, che significa che la struttura ha un solo elemento nel dominio, allora $v(x_3) = v(x_4)$, che è banalmente vero. Tuttavia è impossibile dimostrare $x_1 = x_2 \vdash x_3 = x_4$.
\end{example}

Il guaio nell'esempio precedente sta nel fatto che il simbolo $\models$ intende le premesse come universalmente quantificate, mentre il simbolo $\vdash$ intende le premesse come se $x_1 = x_2$ fosse un assioma, cioè come se fosse una formula chiusa ed $x_1$,$x_2$ fossero due costanti.
Ed un'altra brutta notizia è che il sistema di deduzione naturale non potrebbe funzionare altrimenti. 

\begin{example}[Commutatività dell'uguale]
    Dimostriamo che l'uguaglianza è commutativa, cioè $\vdash \forall x_1 \; \forall x_2 (x_1 = x_2 \to x_2 = x_1)$:
    \begin{align*}
        \text{In}_\forall\frac{\displaystyle\text{In}_\forall\frac{\displaystyle\text{In}_\to\frac{\displaystyle\text{El}_= \frac{\displaystyle \text{Ax}\frac{}{x_1 = x_2 \vdash x_1 = x_2} \; \text{Wk}\frac{\displaystyle \text{In}_{=}\frac{}{\vdash x_1 = x_1}}{x_1 = x_2 \vdash x_1 = x_1} }{x_1 = x_2 \vdash x_2 = x_1}}{x_1 = x_2 \vdash x_1 = x_2 \to x_2 = x_1}}{\forall x_2 (x_1 = x_2 \to x_2 = x_1)}}{\forall x_1 \; \forall x_2 (x_1 = x_2 \to x_2 = x_1)}
    \end{align*}
    dove la $\varphi$ usata in El$_=$ è $\varphi := (x_{17} = x_1)$, per cui $\varphi[x_1/x_{17}] = (x_1 = x_1)$ e $\varphi[x_2/x_{17}] = (x_2 = x_1)$. Se $\vdash$ interpretasse le premesse come universalmente quantificate, il sequente $x_1 = x_2 \vdash x_2 = x_1$ verrebbe letto come ``per ogni valutazione $v$, se $v(x_1)=v(x_2)$ allora $v(x_2)=v(x_1)$''.
    Ma questa è già la lettura semantica della formula chiusa,  $\vdash \forall x_1 \; \forall x_2 (x_1 = x_2 \to x_2 = x_1)$, che vogliamo arrivare a dimostrare. Nel sistema deduttivo, invece, la riga $x_1 = x_2 \vdash x_2 = x_1$ è un passaggio con ipotesi aperta che viene scaricata da In$_\to$ per ottenere $\vdash x_1 = x_2 \to x_2 = x_1$, e solo dopo generalizzata con In$_\forall$.
    In altre parole, partendo da ``il modello ha un solo elemento, allora il modello ha un solo elemento'' non posso arrivare a dimostrare che ``per ogni coppia di elementi del modello se sono uguali in un verso allora sono uguali anche nell'altro verso''.
\end{example}

Per cui, per fare le dimostrazioni nel sistema $\mathsf{ND}$ abbiamo la necessità di trattare gli elementi nella premessa come elementi fissati, per poi poter usare le regole e se lo vogliamo ottenere frasi generalizzate.\\
Morale della favola: non solo $\models$ e $\vdash$ non coincidono, ma non possono nemmeno cambiare le regole per provare a far coincidere $\vdash$, se ci provassi otterrei un sistema che non riesce a dimostrare la commutatività dell'uguale, e di tale sistema ce ne facciamo ben poco.
Per cui va introdotto qualcosa di inelegante per fare in modo che i due simboli $\models,\vdash$ siano uguali, per ora sappiamo che non lo sono se ci sono variabili libere nelle premesse. La vera equivalenza che vogliamo dimostrare è la seguente.

\begin{proposition}[Equivalenza tra $\vdash$ e $\models$]
    Data $T$ una $L$-teoria e $\varphi$ una $L$-formula, vale che $T \vdash \varphi \iff$ \textcolor{purple}{per ogni}
    $L$-struttura $M = (D;i)$ e \textcolor{purple}{per ogni} $v : \Var \to D$ valutazione delle variabili, si ha $M \models \{v\}T \implies M \models \{v\}\varphi$.
\end{proposition}

Questa proposizione ha i due corollari che desideriamo, ovvero l'equivalenza dei due simboli $\vdash$ e $\models$ quando la teoria ha solo formule chiuse.

\begin{corollary}[Teorema di correttezza]
    Sia $T$ una $L$-teoria fatta da $\boxed{\text{formule chiuse}}$, e sia $\varphi$ una $L$-formula, allora:
    \[T \vdash \varphi \implies T \models \varphi\]
\end{corollary}

\begin{corollary}[Teorema di completezza]
    Sia $T$ una $L$-teoria fatta da $\boxed{\text{formule chiuse}}$, e sia $\varphi$ una $L$-formula, allora:
    \[T \models \varphi \implies T \vdash \varphi\]
\end{corollary}
